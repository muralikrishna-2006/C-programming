#include <stdio.h>
int main() {
    int arr[] = {10, 20, 10, 30, 20, 40, 50, 30};
    int size = sizeof(arr) / sizeof(arr[0]);
    int unique[size];
    int newSize = 0;
    int isDuplicate;
    for (int i = 0; i < size; i++) {
        isDuplicate = 0;
        for (int j = 0; j < newSize; j++) {
            if (arr[i] == unique[j]) {
                isDuplicate = 1;
                break;
            }
        }
        if (!isDuplicate) {
            unique[newSize] = arr[i];
            newSize++;
        }
    }
    printf("Array after removing duplicates: ");
    for (int i = 0; i < newSize; i++) {
        printf("%d ", unique[i]);
    }

    return 0;
}
