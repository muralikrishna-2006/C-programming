#include <stdio.h>
int main() {
    int arr[6] = {12, 7, 5, 20, 33, 8};
    int evenSum = 0, oddSum = 0;
    for (int i = 0; i < 6; i++) {
        if (arr[i] % 2 == 0) {
            evenSum += arr[i];
        } else {
            oddSum += arr[i];
        }
    }
    printf("Sum of even elements: %d\n", evenSum);
    printf("Sum of odd elements: %d\n", oddSum);

    return 0;
}

Output

Sum of even elements: 40
Sum of odd elements: 45
