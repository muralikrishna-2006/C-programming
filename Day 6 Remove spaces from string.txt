#include <stdio.h>
int main() {
    char str[] = "Remove all spaces";
    char result[100]; 
    int i = 0, j = 0;
    while (str[i] != '\0') {
        if (str[i] != ' ') {
            result[j++] = str[i];
        }
        i++;
    }
    result[j] = '\0';
    printf("Original: %s\n", str);
    printf("Without spaces: %s\n", result);

    return 0;
}
